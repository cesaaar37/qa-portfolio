Приклад порівняння данних, що відправляються у запиті з данними, що приходять у відповіді на запит.
Відображення значення через console.log. Використання Snippets

var resJson = pm.response.json();
var req = JSON.parse(pm.request.body.raw);

var resName = resJson.name;
console.log("rws: " + resName);

pm.collectionVariables.set("Name_pj_res", req.name);
console.log("Name_pj_res: " + pm.collectionVariables.get("Name_pj_res"));


pm.test("Test project name", function () {
    pm.expect(resJson.name).to.eql(pm.collectionVariables.get("Name_pj_res")); //порівняння через змінну колекції
    pm.expect(req.name).to.eql(resJson.name); //порівняння через використання ключів запит\відповідь без створення collectionVariables
});
